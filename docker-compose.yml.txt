version: '3.8'

services:
  # Backend API Service
  backend:
    build: ./backend
    container_name: task-backend
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=postgresql://postgres:postgres@postgres:5432/taskdb
      - REDIS_URL=redis://redis:6379/0
      - SECRET_KEY=your-super-secret-key-change-this-in-production
      - SMTP_HOST=smtp.gmail.com
      - SMTP_PORT=587
      - SMTP_USER=your-email@gmail.com
      - SMTP_PASSWORD=your-app-password
    depends_on:
      - postgres
      - redis
    volumes:
      - ./backend:/app
      - /app/__pycache__
    networks:
      - task-network
    restart: unless-stopped
    command: uvicorn app.main:app --host 0.0.0.0 --port 8000 --reload

  # Frontend React Service
  frontend:
    build: ./frontend
    container_name: task-frontend
    ports:
      - "3000:3000"
    environment:
      - VITE_API_URL=http://localhost:8000
    volumes:
      - ./frontend:/app
      - /app/node_modules
    depends_on:
      - backend
    networks:
      - task-network
    restart: unless-stopped
    command: npm run dev -- --host 0.0.0.0

  # PostgreSQL Database Service
  postgres:
    image: postgres:15-alpine
    container_name: task-postgres
    environment:
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=postgres
      - POSTGRES_DB=taskdb
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./backend/alembic:/docker-entrypoint-initdb.d
    networks:
      - task-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis Cache Service
  redis:
    image: redis:7-alpine
    container_name: task-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    networks:
      - task-network
    restart: unless-stopped
    command: redis-server --appendonly yes

  # Celery Worker Service
  celery:
    build: ./backend
    container_name: task-celery
    command: celery -A app.tasks worker --loglevel=info
    environment:
      - DATABASE_URL=postgresql://postgres:postgres@postgres:5432/taskdb
      - REDIS_URL=redis://redis:6379/0
      - SECRET_KEY=your-super-secret-key-change-this-in-production
    depends_on:
      - redis
      - postgres
    volumes:
      - ./backend:/app
    networks:
      - task-network
    restart: unless-stopped

  # Celery Beat (Scheduler) Service
  celery-beat:
    build: ./backend
    container_name: task-celery-beat
    command: celery -A app.tasks beat --loglevel=info
    environment:
      - DATABASE_URL=postgresql://postgres:postgres@postgres:5432/taskdb
      - REDIS_URL=redis://redis:6379/0
    depends_on:
      - redis
      - postgres
      - celery
    volumes:
      - ./backend:/app
    networks:
      - task-network
    restart: unless-stopped

  # PgAdmin (Optional - for database management)
  pgadmin:
    image: dpage/pgadmin4
    container_name: task-pgadmin
    environment:
      - PGADMIN_DEFAULT_EMAIL=admin@admin.com
      - PGADMIN_DEFAULT_PASSWORD=admin
    ports:
      - "5050:80"
    depends_on:
      - postgres
    networks:
      - task-network
    restart: unless-stopped

networks:
  task-network:
    driver: bridge

volumes:
  postgres_data:
    driver: local
  redis_data:
    driver: local
